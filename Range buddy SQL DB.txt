create table Engagement (
		PRIMARY KEY (ENGAGEMENT_ID),
		ENGAGEMENT_ID INTEGER GENERATED BY DEFAULT AS IDENTITY,
		CALIBER INTEGER NOT NULL,
		GRAINS INTEGER NOT NULL,
		VELOCITY INTEGER NOT NULL,
		WIND_DIR INTEGER,
		WIND_SPEED INTEGER,
		CLICK_VALUE INTEGER NOT NULL,
		ZERO INTEGER NOT NULL,
		WINDAGE_ADJ INTEGER,
		ELEVATION_ADJ INTEGER
		TARGET_STYLE_ID INTEGER NOT NULL CONSTRAINT TARGET_STYLE_FK REFERENCES TARGET_STYLE
		)
	
CREATE TABLE SHOT (
		PRIMARY KEY (SHOT_ID)
		SHOT_ID INTEGER NOT NULL GENERATED BY DEFAULT AS IDENTITY,
		X_COORDINATES INTEGER NOT NULL,
		Y_COORDINATES INTEGER NOT NULL,
		SEQUENCE INTEGER NOT NULL,
		ENGAGEMENT_ID INTEGER NOT NULL CONSTRAINT ENGAGEMENT_ID_FK REFERENCES ENGAGEMENT,
		
		)

CREATE TABLE TARGET_STYLE (
		PRIMARY KEY (TARGET_STYLE_ID)
		TARGET_STYLE_ID INTEGER NOT NULL GENERATED BY DEFAULT AS IDENTITY,
		STYLE VARCHAR(255) NULL,
		HEIGHT INTEGER NOT NULL,
		WIDTH INTEGER NOT NULL
		)
